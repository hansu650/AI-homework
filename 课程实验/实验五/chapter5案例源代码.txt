【例5-1】使用传统RNN生成前 100 个斐波那契数。
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np

# 1. 生成斐波那契数列数据
def generate_fibonacci(n_terms):
    fib = [0, 1]
    for i in range(2, n_terms):
        fib.append(fib[-1] + fib[-2])
    return fib[:n_terms]

# 生成 100 个数字用于训练和测试
fib_seq = generate_fibonacci(100)
print("斐波那契数列前 10 项:", fib_seq[:10])

# 2. 构造训练样本
# 使用滑动窗口：输入 (x_t, x_{t+1})，输出 x_{t+2}
seq_len = 2  # 输入序列长度（恰好是递推所需的前两项）
input_size = 1  # 每个时间步的特征维度（标量）
X, y = [], []
for i in range(len(fib_seq) - seq_len - 1):
    X.append(fib_seq[i:i + seq_len])  # 长度为 2 的列表
    y.append(fib_seq[i + seq_len])  # 下一项

# 数据集形状X: (samples, seq_len, input_size)，y: (samples, 1)
X = torch.tensor(X, dtype=torch.float32).unsqueeze(-1)  
y = torch.tensor(y, dtype=torch.float32).unsqueeze(-1)  # 

# 划分训练集和验证集 (80% 训练, 20% 验证)
split = int(0.8 * len(X))
X_train, X_val = X[:split], X[split:]
y_train, y_val = y[:split], y[split:]

# 3. 定义 RNN 模型
class FibonacciRNN(nn.Module):
    def __init__(self, input_size=1, hidden_size=32, num_layers=1, output_size=1):
        super(FibonacciRNN, self).__init__()
        self.rnn = nn.RNN(input_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, output_size)

    def forward(self, x):
        # x shape: (batch, seq_len, input_size)
        out, _ = self.rnn(x)  # out: (batch, seq_len, hidden_size)
        last_out = out[:, -1, :]  # 取最后一个时间步的输出
        pred = self.fc(last_out)  # (batch, output_size)
        return pred

# 创建模型
model = FibonacciRNN(input_size=1, hidden_size=32, num_layers=1)

# 4. 训练模型
criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)
num_epochs = 500
batch_size = 16

# 创建 DataLoader
train_dataset = torch.utils.data.TensorDataset(X_train, y_train)
train_loader=torch.utils.data.DataLoader(train_dataset,batch_size=batch_size,shuffle=True)

val_dataset = torch.utils.data.TensorDataset(X_val, y_val)
val_loader = torch.utils.data.DataLoader(val_dataset, batch_size=batch_size)

for epoch in range(num_epochs):
    model.train()
    total_train_loss = 0
    for batch_X, batch_y in train_loader:
        optimizer.zero_grad()
        outputs = model(batch_X)
        loss = criterion(outputs, batch_y)
        loss.backward()
        optimizer.step()
        total_train_loss += loss.item() * batch_X.size(0)
    avg_train_loss = total_train_loss / len(X_train)

    # 验证
    model.eval()
    total_val_loss = 0
    with torch.no_grad():
        for batch_X, batch_y in val_loader:
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            total_val_loss += loss.item() * batch_X.size(0)
    avg_val_loss = total_val_loss / len(X_val)

    if (epoch + 1) % 100 == 0:
        print(f"Epoch [{epoch + 1}/{num_epochs}], Train Loss: {avg_train_loss:.6f}, Val Loss: {avg_val_loss:.6f}")

# 5. 测试预测
def predict_next(a, b):
    """输入前两项 a, b，预测下一项"""
    model.eval()
    with torch.no_grad():
        input_tensor = torch.tensor([[a, b]], dtype=torch.float32).unsqueeze(-1) 
        pred = model(input_tensor).item()
        return round(pred)  # 四舍五入取整

# 预测更大数值
print("\n--- 泛化测试（未出现在训练集中）---")
print(f"输入 (13, 21) → 预测: {predict_next(13, 21)}, 真实: 34")
print(f"输入 (21, 34) → 预测: {predict_next(21, 34)}, 真实: 55")
print(f"输入 (34, 55) → 预测: {predict_next(34, 55)}, 真实: 89")
